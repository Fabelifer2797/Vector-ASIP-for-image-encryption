MOVI RS1, 0X100 // UBICACION DONDE SE GUARDAN LOS PIXELES
MOVI RS2, 307200 // (TOTAL DE PIXELES DE LA IMAGEN IXJ, ES UN EJEMPLO)
MOVI RS3, 0 // CONTADOR
MOVI RS11, 8 // PARA SUMA DE CONTADOR DE PIXELES
MOVI RS5, 0X90000 // UBICACION DE LOS PIXELES
MOVI RS10, 64 // VALOR DE CUANTO SE VAN A MOVER EN MEMORIA
MOVI RS6, 127 // CANTIDAD DE BITS Round 1
MOVI RS7, 34 // LLAVE DEL XOR Round 1
MOVI RS8, 55 // CANTIDAD DE BITS Round 2
MOVI RS9, 10 // LLAVE DEL XOR Round 2
OBT_BYTES :
MOVRS RS4, RS5 // OBTIENE LA UBICACION DE LOS PIXELES
LDRV RV1, RS4 // RV1 OBTIENE LO QUE HAY EN RS4
APLIC_ENCRYPT :
RCRVS RV2, RV1, RS6 // APLICA EL CORRIMIENTO AL VECTOR Y SE GUARDA EN RV2
XORVS RV3, RV2, RS7 // APLICA EL XOR AL VECTOR Y SE GUARDA EN RV2
RCRVS RV4, RV3, RS8 // APLICA EL CORRIMIENTO AL VECTOR Y SE GUARDA EN RV2
XORVS RV5, RV4, RS9 // APLICA EL XOR AL VECTOR Y SE GUARDA EN RV2
STRV RS1, RV5 // GUARDA EL VECTOR EN 0X100
ADDS RS1, RS1, RS10 // LA MEMORIA DE ALMACENAMIENTO SE MUEVE 64 BITS
ADDS RS5, RS5, RS10 // LA MEMORIA DE BUSQUEDA SE MUEVE 64 BITS
ADDS RS3, RS3, RS11 // SE LE SUMA 8 AL CONTADOR DE PIXELES
CMPR RS2, RS3 // COMPARA SI 307200 ES MAYOR AL CONTADOR
JGT OBT_BYTES // SI ES MAYOR MUEVASE A OBT_BYTES
JMP END // SINO TERMINE
END :
